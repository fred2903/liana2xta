create automaton Flower
{
    clocks { x1, x2, x3, x4, y; }
    actions { a; }
    locations {
        Start <ini: T>,
        Goal  <ini: F>;
    }
    transitions {
        (Start, a, [(x4, ==, 4)], [x4, x3, x2], Start),
        (Start, a, [(x3, ==, 3)], [x3], Start),
        (Start, a, [(x2, ==, 2)], [x2], Start),
        (Start, a, [(x1, ==, 1)], [x1], Start),
        (Start, a, [(x1, <, 0), (x2, ==, 0), (x3, ==, 0), (x4, ==, 0), (y, >=, 1)], [], Goal);
    }
}
